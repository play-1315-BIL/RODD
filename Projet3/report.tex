\begin{align*}
    \min_{x, p, t} &\sum_{i \leq M} \sum_{j \leq t(i)} p_{ij}\\
    s.t.   & \log(t_{ij}) \geq \sum_{k \in K_{ij}^1} x_k \log\left(\frac{1}{2}\right)\\
           & p_{ij} \geq t_{ij} - \sum_{k \in K_{ij}^2} x_k\\
           & \sum_{k \in \mathcal{M}} x_k = P\\
           & \sum_{k \in \mathcal{F}} x_k = P\\
           & x_k \in \mathbb{N}\\
           & p_{ij} \geq 0
\end{align*}
